1/**
2 * @file TestAbleButton.ino Test the Arduino Button Library Extension (ABLE).
3 * This program will periodically output "Looking OK..." to the Serial port. As
4 * you press buttons connected to pins 2 and 3 it will assert the button
5 * settings in the callback fuctions and in the main loop.
6 *
7 * The TestAbleButton program is split (by design) into multiple files to ensure
8 * AbleButtons works in multi-module programs. A common Config.h header file
9 * declares common elements shared across these modules.

63/**
64 * Control TestAbleButton. Called repeatedly in a loop.
65 */
66void loop() {
67 static unsigned long ms = millis(); ///< Timer to preiodically print OK message.
68
69 if(millis() - ms > 5000) {
70 Serial.println(F("Looking OK..."));
71 ms = millis();
72 }
73
74 btnList.handle(); // Handle features of the buttons (debouncing, clicks & callbacks).
75
76#if TESTABLE_CLASS < TESTABLE_CLICKER
77 led = btnList.anyPressed(); // Display pressed activity to LED.
78#else
79 if(btnList.anyClicked()) { // Display clicked activity to LED.
80 led = !led;
81 }
82#endif
83 digitalWrite(LED_BUILTIN, led); // Update the LED.
84
85 checkButtonListIntegrity(); // Check ButtonList::any* and ButtonList::all* states...
86
87 for(int i = 0; i < NUM_BUTTONS; ++i) {
88 Button *btn = btns[i];
89 ButtonState &state(btnState[i]);
90
91 checkButtonIntegrity(btn, state);
92
93# if TESTABLE_CALLBACK
94 // Check state when button pressed event.
95 if(state.wasPressed) {
96 checkButtonJustPressed(btn);
97 state.wasPressed = false; // Clear most recent callback signal.
98 }
99
100 // Check state when button released event.
101 if(state.wasReleased) {
102 checkButtonJustReleased(btn);
103 state.wasReleased = false; // Clear most recent callback signal.
104 }
105
106 // Check state when button held event.
107 if(state.wasHeld) {
108 checkButtonJustHeld(btn);
109 state.wasHeld = false; // Clear most recent callback signal.
110 }
111
112 // Check state when button idle event.
113 if(state.wasIdle) {
114 checkButtonJustIdle(btn);
115 state.wasIdle = false; // Clear most recent release callback signal.
116 }
117# endif
118
119 checkButtonListIntegrity();
120 displayButtonChanges(i);
121 }
122
123 // Reset clicks/double-clicks and check.
124# if TESTABLE_CLASS >= TESTABLE_CLICKER
125 assert(btnList.anyClicked() == btnList.resetClicked());
126 assert(btnList.anyClicked() == false);

135}
