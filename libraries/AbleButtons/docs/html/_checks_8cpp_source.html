178/**
179 * Check integrity of Button invariants (always hold true).
180 *
181 * @param btn The button to check.
182 * @param state The tracked state of the button for comparison.
183 */
184void checkButtonIntegrity(Button *btn, ButtonState &state) {
185 // Basic checks...
186 if(btn->isPressed()) {
187 // Given a pressed button...
188 if(!state.isPressed) {
189 // ...when it is just pressed, then...
190 checkButtonJustPressed(btn);
191 } else {
192# if TESTABLE_CLASS >= TESTABLE_BUTTON
193 // ...when it is still pressed, then it...
194 // assert(btn->isHeld() == false); // ...*might* be held.
195 assert(btn->isIdle() == false); // ...cannot be idle.
196# endif
197
198# if TESTABLE_CLASS >= TESTABLE_CLICKER
199 assert(btn->isClicked() == false); // ...cannot be clicked.
200# endif
201
202# if TESTABLE_CLASS >= TESTABLE_DOUBLECLICKER
203 assert(btn->isDoubleClicked() == false); // ...cannot double-clicked.
204# endif
205 }
206 } else {
207 // Given an unpressed button...
208 if(state.isPressed) {
209 // ...when it is just released, then...
210 checkButtonJustReleased(btn);
211 } else {
212# if TESTABLE_CLASS >= TESTABLE_BUTTON
213 // ...when it is still released, then it...
214 assert(btn->isHeld() == false); // ...cannot be held.
215 // assert(btn->isIdle() == false); // ...*might* be idle.
216# endif
217
218# if TESTABLE_CLASS >= TESTABLE_CLICKER
219 assert(btn->isClicked() == false); // ...cannot be clicked.
220# endif
221
222# if TESTABLE_CLASS >= TESTABLE_DOUBLECLICKER
223 assert(btn->isDoubleClicked() == false); // ...cannot double-clicked.
224# endif
225 }
226 }
227
228
229# if TESTABLE_CLASS >= TESTABLE_BUTTON
230 if(!state.isHeld && btn->isHeld()) {
231 // Given a button that has just become held, then...
232 checkButtonJustHeld(btn);
233 }
234
235 if(!state.isIdle && btn->isIdle()) {
236 // Given a button that has just become idle, then...
237 checkButtonJustIdle(btn);
238 }
239# endif
240
241 // Clickable checks...
242# if TESTABLE_CLASS >= TESTABLE_CLICKER
243 if(!state.isClicked && btn->isClicked()) {
244 // Given a button that has just been clicked, then...
245 checkButtonJustClicked(btn);
246 }
247# endif
248
249 // Double clicker checks...
250# if TESTABLE_CLASS >= TESTABLE_DOUBLECLICKER
251 if(!state.isDoubleClicked && btn->isDoubleClicked()) {
252 // given a button that has just been double-clicked, then...
253 checkButtonJustDoubleClicked(btn);
254 }
255# endif
256}

259/**
260 * Check integrity of ButtonList invariants (always hold true).
261 */
262void checkButtonListIntegrity() {
263 // ButtonList::all*() accessors...
264 assert(btnList.allPressed() == (btnA.isPressed() && btnB.isPressed()));
265
266# if TESTABLE_CLASS >= TESTABLE_BUTTON
267 // Because isHeld and isIdle use millis() timings, the state can change
268 // whilst checking ButtonList::allHeld/Idle and each Button:isHeld/Idle. So
269 // checking each twice ensures if the first check is false, it tries once
270 // more. Only if both comparisons are false means we have a problem...
271 assert(
272 (btnList.allHeld() == (btnA.isHeld() && btnB.isHeld())) ||
273 (btnList.allHeld() == (btnA.isHeld() && btnB.isHeld()))
274 );
275 assert(
276 (btnList.allIdle() == (btnA.isIdle() && btnB.isIdle())) ||
277 (btnList.allIdle() == (btnA.isIdle() && btnB.isIdle()))
278 );
279# endif
280
281# if TESTABLE_CLASS >= TESTABLE_CLICKER
282 assert(btnList.allClicked() == (btnA.isClicked() && btnB.isClicked()));
283# endif
284
285# if TESTABLE_CLASS >= TESTABLE_DOUBLECLICKER
286 assert(btnList.allDoubleClicked() == (btnA.isDoubleClicked() && btnB.isDoubleClicked()));
287# endif
288
289 // ButtonList::any*() accessors...
290 assert(btnList.anyPressed() == (btnA.isPressed() || btnB.isPressed()));
291
292# if TESTABLE_CLASS >= TESTABLE_BUTTON
293 // Because isHeld and isIdle use millis() timings, the state can change
294 // whilst checking ButtonList::anyHeld/Idle and each Button:isHeld/Idle. So
295 // checking each twice ensures if the first check is false, it tries once
296 // more. Only if both comparisons are false means we have a problem...
297 assert(
298 (btnList.anyHeld() == (btnA.isHeld() || btnB.isHeld())) ||
299 (btnList.anyHeld() == (btnA.isHeld() || btnB.isHeld()))
300 );
301 assert(
302 (btnList.anyIdle() == (btnA.isIdle() || btnB.isIdle())) ||
303 (btnList.anyIdle() == (btnA.isIdle() || btnB.isIdle()))
304 );
305# endif
306
307# if TESTABLE_CLASS >= TESTABLE_CLICKER
308 assert(btnList.anyClicked() == (btnA.isClicked() || btnB.isClicked()));
309# endif
310
311# if TESTABLE_CLASS >= TESTABLE_DOUBLECLICKER
312 assert(btnList.anyDoubleClicked() == (btnA.isDoubleClicked() || btnB.isDoubleClicked()));
313# endif
314}

322void displayButtonChanges(int index) {
323 Button *btn = btns[index];
324 ButtonState &state(btnState[index]);
325
326 // Basic checks...

333 if(state.isPressed) {
334 Serial << F("Button btns[") << index << F("] (btn") << (char)('A' + index) << F(") released") << endl;

336 state.isPressed = false;
337 }
338
339# if TESTABLE_CLASS >= TESTABLE_BUTTON
340 if(btn->isHeld()) {

343 }
344 state.isHeld = true;
345 } else {
346 if(state.isHeld) {
347 Serial << F("Button btns[") << index << F("] (btn") << (char)('A' + index) << F(") un-held") << endl;
348 }
349 state.isHeld = false;
350 }
351
352 if(btn->isIdle()) {
353 if(!state.isIdle) {
354 Serial << F("Button btns[") << index << F("] (btn") << (char)('A' + index) << F(") idle") << endl;
355 }
356 state.isIdle = true;
357 } else {
358 if(state.isIdle) {
359 Serial << F("Button btns[") << index << F("] (btn") << (char)('A' + index) << F(") un-idle") << endl;
360 }
361 state.isIdle = false;
362 }
363# endif
364
365 // Clickable checks...
366# if TESTABLE_CLASS >= TESTABLE_CLICKER
367 if(!state.isClicked && btn->isClicked()) {
368 Serial << F("Button btns[") << index << F("] (btn") << (char)('A' + index) << F(") clicked") << endl;
369 state.isClicked = true;
370 } else if(state.isClicked) {
371 state.isClicked = false;
372 }
373# endif
374
375 // Double clicker checks...
376# if TESTABLE_CLASS >= TESTABLE_DOUBLECLICKER
377 if(!state.isDoubleClicked && btn->isDoubleClicked()) {
378 Serial << F("Button btns[") << index << F("] (btn") << (char)('A' + index) << F(") double-clicked") << endl;
379 state.isDoubleClicked = true;
380 } else {
381 state.isDoubleClicked = false;
382 }
383# endif
